def modify_list(l):
    for i in range(-1,-len(l),-1 ):
        try:    
            if int(l[i]) % 2 != 0:
                l.remove(l[i])
                i += 1
        except IndexError:        
            for i in range(len(l)):
                l[i] = int((l[i]) / 2)
            return
lst = [4, 1, 1, 2, 2, 2, 3, 4, 5, 6, 7]
modify_list(lst)
print(lst)
