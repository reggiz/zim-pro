unl: добавить столбец в file

Created четверг 18 Март 2021
@python
выбор файла по кнопке 

# coding=utf-8
import os
import tkinter as tk
from tkinter.filedialog import askopenfilename
from tkinter.messagebox import showerror

ADD = "0|" #добавляемое выражение

def zero_to_end(src_filename: str) -> str:
    dst_file = src_filename
    os.rename(src_filename, src_filename + '.bak')
    with open(dst_file, "w") as wf:
        with open(src_filename + '.bak', "r") as f:
            lines = f.read().splitlines()
            for line in lines:
                wf.write(f"{line}{ADD}\n")

    return f"Success write file {dst_file}"



def work() -> None:
	#выбор файла
    filename = askopenfilename()
    if filename:
        print(zero_to_end(filename))
    else:
        showerror("Wrong File", f"{filename}")


if __name__ == "__main__":

    root = tk.Tk()
    tk.Button(root, text="Open file", command=work).pack()
    root.mainloop()
